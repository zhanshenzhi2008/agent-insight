### 2.2 DynamicDatasourceManager（多数据源连接池）

核心实现，支持 MySQL / PostgreSQL / MongoDB，默认端口从环境变量读取：

```java
package com.llm.insight.explorer.engine;

import com.llm.insight.explorer.document.InsightDatasource;
import com.zaxxer.hikari.HikariConfig;
import com.zaxxer.hikari.HikariDataSource;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.mongodb.core.MongoTemplate;
import org.springframework.stereotype.Component;

import javax.sql.DataSource;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

/**
 * 动态数据源连接池管理器。
 * 按 datasourceKey 缓存连接池，用到时懒加载，不用时不占资源。
 * <p>
 * 默认端口支持 env 覆盖，优先级：
 * 1. ConnectionConfig.port（非 null）
 * 2. 环境变量 POSTGRES_PORT / MYSQL_PORT / MONGODB_PORT
 * 3. 内置默认值
 */
@Slf4j
@Component
public class DynamicDatasourceManager {

    private static final int DEFAULT_MYSQL_PORT = 3306;
    private static final int DEFAULT_POSTGRES_PORT = 5433;
    private static final int DEFAULT_MONGODB_PORT = 27017;

    private final Map<String, DataSource> sqlDataSources = new ConcurrentHashMap<>();
    private final Map<String, MongoTemplate> mongoTemplates = new ConcurrentHashMap<>();
    private final Map<String, InsightDatasource> datasourceCache = new ConcurrentHashMap<>();

    public DataSource getSqlDataSource(InsightDatasource ds) {
        return sqlDataSources.computeIfAbsent(ds.getDatasourceKey(), key -> {
            log.info("创建 MySQL/PG 连接池: datasourceKey={}", key);
            HikariConfig config = buildHikariConfig(ds);
            return new HikariDataSource(config);
        });
    }

    public MongoTemplate getMongoTemplate(InsightDatasource ds) {
        return mongoTemplates.computeIfAbsent(ds.getDatasourceKey(), key -> {
            log.info("创建 MongoTemplate: datasourceKey={}", key);
            return createMongoTemplate(ds);
        });
    }

    public void evict(String datasourceKey) {
        DataSource ds = sqlDataSources.remove(datasourceKey);
        if (ds instanceof HikariDataSource hds) {
            hds.close();
        }
        MongoTemplate mt = mongoTemplates.remove(datasourceKey);
        datasourceCache.remove(datasourceKey);
    }

    private HikariConfig buildHikariConfig(InsightDatasource ds) {
        InsightDatasource.ConnectionConfig cc = ds.getConnectionConfig();

        HikariConfig config = new HikariConfig();
        config.setPoolName("insight-pool-" + ds.getDatasourceKey());
        config.setMaximumPoolSize(cc.getConnectionPoolSize() != null ? cc.getConnectionPoolSize() : 5);
        config.setMinimumIdle(1);
        config.setConnectionTimeout(
                cc.getConnectionTimeout() != null ? (long) cc.getConnectionTimeout() : 30000);

        String url = buildJdbcUrl(ds.getDatasourceType(), cc);
        config.setJdbcUrl(url);
        config.setUsername(cc.getUsername());
        config.setPassword(cc.getPassword());
        config.setDriverClassName(getDriverClass(ds.getDatasourceType()));

        return config;
    }

    private String buildJdbcUrl(String type, InsightDatasource.ConnectionConfig cc) {
        String host = cc.getHost();
        int port = cc.getPort() != null ? cc.getPort() : getDefaultPort(type);
        String db = cc.getDatabase();
        StringBuilder sb = new StringBuilder();

        if ("MYSQL".equalsIgnoreCase(type)) {
            sb.append("jdbc:mysql://").append(host).append(":").append(port)
                    .append("/").append(db)
                    .append("?useSSL=false")
                    .append("&allowPublicKeyRetrieval=true")
                    .append("&serverTimezone=")
                    .append(cc.getExtraParams().getOrDefault("serverTimezone", "Asia/Shanghai"))
                    .append("&connectTimeout=").append(cc.getConnectionTimeout() != null ? cc.getConnectionTimeout() : 30000);
        } else if ("POSTGRESQL".equalsIgnoreCase(type)) {
            sb.append("jdbc:postgresql://").append(host).append(":").append(port)
                    .append("/").append(db)
                    .append("?loginTimeout=").append((cc.getConnectionTimeout() != null ? cc.getConnectionTimeout() : 30));
        }

        return sb.toString();
    }

    private String getDriverClass(String type) {
        return switch (type.toUpperCase()) {
            case "MYSQL" -> "com.mysql.cj.jdbc.Driver";
            case "POSTGRESQL" -> "org.postgresql.Driver";
            default -> throw new IllegalArgumentException("Unsupported SQL type: " + type);
        };
    }

    private int getDefaultPort(String type) {
        return switch (type.toUpperCase()) {
            case "MYSQL" -> getEnvInt("MYSQL_PORT", DEFAULT_MYSQL_PORT);
            case "POSTGRESQL" -> getEnvInt("POSTGRES_PORT", DEFAULT_POSTGRES_PORT);
            default -> DEFAULT_MYSQL_PORT;
        };
    }

    private int getEnvInt(String key, int fallback) {
        String val = System.getenv(key);
        if (val != null && !val.isBlank()) {
            try {
                return Integer.parseInt(val);
            } catch (NumberFormatException ignored) {}
        }
        return fallback;
    }

    private MongoTemplate createMongoTemplate(InsightDatasource ds) {
        InsightDatasource.ConnectionConfig cc = ds.getConnectionConfig();
        StringBuilder uri = new StringBuilder("mongodb://");
        if (cc.getUsername() != null && !cc.getUsername().isEmpty()) {
            uri.append(cc.getUsername()).append(":").append(cc.getPassword()).append("@");
        }
        uri.append(cc.getHost()).append(":")
                .append(cc.getPort() != null ? cc.getPort() : getEnvInt("MONGODB_PORT", DEFAULT_MONGODB_PORT))
                .append("/").append(cc.getDatabase());
        String authSource = cc.getExtraParams() != null ? cc.getExtraParams().get("authSource") : null;
        if (authSource != null) uri.append("?authSource=").append(authSource);

        var mongoClient = com.mongodb.client.MongoClients.create(uri.toString());
        return new MongoTemplate(mongoClient, cc.getDatabase());
    }
}
```

```java
package com.llm.insight.explorer.ai;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.client.SystemAdvisors;
import org.springframework.ai.chat.client.advisor.MessageChatMemoryAdvisor;
import org.springframework.ai.chat.memory.InMemoryChatMemory;
import org.springframework.stereotype.Service;

@Slf4j
@Service
@RequiredArgsConstructor
public class AiChatService {

    private final AiProperties aiProperties;
    private final ChatClient openAiChatClient;
    private final ChatClient deepseekChatClient;
    private final ChatClient ollamaChatClient;

    private final InMemoryChatMemory chatMemory = new InMemoryChatMemory();

    public String analyzeColumn(String columnName, Object sampleValues) {
        if (!aiProperties.isEnabled() || !aiProperties.isColumnAnalysisEnabled()) {
            return null;
        }

        String prompt = String.format("""
            分析数据库列 '%s' 的数据特征。
            样本数据: %s
            请返回 JSON 格式，包含：
            - description: 列的中文描述
            - suggestedRenderType: 建议渲染类型（text/link/tag/money/date/boolean）
            - insights: 关键发现（最多 3 条）
            """, columnName, sampleValues);

        return callAi(prompt);
    }

    public String nlToSql(String naturalLanguage, String tableSchema) {
        if (!aiProperties.isEnabled() || !aiProperties.isNlQueryEnabled()) {
            return null;
        }

        String prompt = String.format("""
            将以下自然语言转换为 SQL 查询语句。
            表结构：%s
            自然语言：%s
            只返回 SQL 语句本身，不要其他解释。
            """, tableSchema, naturalLanguage);

        return callAi(prompt);
    }

    private String callAi(String prompt) {
        ChatClient chatClient = resolveClient();
        if (chatClient == null) {
            log.warn("AI provider '{}' not available", aiProperties.getProvider());
            return null;
        }

        try {
            return chatClient.prompt()
                    .advisors(new MessageChatMemoryAdvisor(chatMemory, "default"))
                    .user(prompt)
                    .call()
                    .content();
        } catch (Exception e) {
            log.error("AI call failed: {}", e.getMessage());
            return null;
        }
    }

    private ChatClient resolveClient() {
        return switch (aiProperties.getProvider().toLowerCase()) {
            case "deepseek" -> deepseekChatClient;
            case "ollama" -> ollamaChatClient;
            case "openai" -> openAiChatClient;
            default -> openAiChatClient;
        };
    }
}
```

### 2.5 QueryRouter（查询路由）

```java
package com.llm.insight.explorer.engine;

import com.llm.insight.explorer.document.InsightDatasource;
import com.llm.insight.explorer.document.InsightTableConfig;
import com.llm.insight.explorer.service.ConfigService;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

@Component
@RequiredArgsConstructor
public class QueryRouter {

    private final ConfigService configService;
    private final DynamicDatasourceManager datasourceManager;

    public QueryExecutor route(String tableKey) {
        InsightTableConfig tableConfig = configService.getTableConfig(tableKey);
        if (tableConfig == null) {
            throw new IllegalArgumentException("Table config not found: " + tableKey);
        }

        InsightDatasource ds = configService.getDatasource(tableConfig.getDatasourceKey());
        if (ds == null) {
            throw new IllegalArgumentException("Datasource not found: " + tableConfig.getDatasourceKey());
        }

        return switch (ds.getDatasourceType().toUpperCase()) {
            case "MYSQL", "POSTGRESQL" -> new SqlQueryExecutor(
                    datasourceManager.getSqlDataSource(ds), ds.getDatasourceType());
            case "MONGODB" -> new MongoQueryExecutor(
                    datasourceManager.getMongoTemplate(ds));
            default -> throw new UnsupportedOperationException(
                    "Unsupported datasource type: " + ds.getDatasourceType());
        };
    }
}
```

### 2.6 SqlQueryExecutor（SQL 查询执行器）

```java
package com.llm.insight.explorer.engine;

import com.llm.insight.explorer.document.InsightDatasource;
import com.llm.insight.explorer.dto.QueryRequest;
import com.llm.insight.explorer.dto.QueryResponse;
import lombok.RequiredArgsConstructor;
import org.springframework.jdbc.core.JdbcTemplate;
import org.springframework.jdbc.core.namedparam.NamedParameterJdbcTemplate;

import javax.sql.DataSource;
import java.util.*;

@RequiredArgsConstructor
public class SqlQueryExecutor implements QueryExecutor {

    private final DataSource dataSource;
    private final String datasourceType;

    @Override
    public QueryResponse execute(QueryRequest request, InsightDatasource ds,
                                 com.llm.insight.explorer.document.InsightTableConfig tableConfig) {
        JdbcTemplate jdbc = new JdbcTemplate(dataSource);
        NamedParameterJdbcTemplate npJdbc = new NamedParameterJdbcTemplate(jdbc);

        String tableName = tableConfig.getTableName();
        Map<String, Object> params = new HashMap<>();

        // 构建 SELECT 子句
        String selectClause = request.columns().isEmpty()
                ? "*"
                : String.join(", ", request.columns());

        // 构建 WHERE 子句
        StringBuilder whereClause = new StringBuilder();
        if (!request.filters().isEmpty()) {
            for (var filter : request.filters()) {
                if (!whereClause.isEmpty()) {
                    whereClause.append(" AND ");
                }
                whereClause.append(filter.column())
                        .append(" ")
                        .append(toSqlOperator(filter.operator()))
                        .append(" :")
                        .append(filter.column().replace(".", "_"));
                params.put(filter.column().replace(".", "_"), filter.value());
            }
        }

        // 构建 ORDER BY 子句
        StringBuilder orderByClause = new StringBuilder();
        if (!request.sorts().isEmpty()) {
            orderByClause.append(" ORDER BY ");
            for (var sort : request.sorts()) {
                if (orderByClause.length() > " ORDER BY ".length()) {
                    orderByClause.append(", ");
                }
                orderByClause.append(sort.column()).append(" ")
                        .append(sort.order().name());
            }
        }

        // 构建完整 SQL
        String sql = "SELECT " + selectClause + " FROM " + tableName;
        if (!whereClause.isEmpty()) {
            sql += " WHERE " + whereClause;
        }
        sql += orderByClause;

        // COUNT 查询
        String countSql = "SELECT COUNT(*) FROM " + tableName;
        if (!whereClause.isEmpty()) {
            countSql += " WHERE " + whereClause;
        }
        long totalCount = npJdbc.queryForObject(countSql, params, Long.class);

        // 分页（统一使用 LIMIT/OFFSET，兼容 MySQL 和 PostgreSQL）
        sql += " LIMIT :limit OFFSET :offset";
        params.put("limit", request.pageSize());
        params.put("offset", request.page() * request.pageSize());

        // 执行查询
        List<Map<String, Object>> rows = npJdbc.queryForList(sql, params);

        return new QueryResponse(
                rows,
                List.of(), // columns 由 ConfigService 补充
                totalCount,
                request.page(),
                request.pageSize(),
                (int) Math.ceil((double) totalCount / request.pageSize()),
                System.currentTimeMillis()
        );
    }

    private String toSqlOperator(com.llm.insight.explorer.dto.FilterOperator op) {
        return switch (op) {
            case EQ -> "=";
            case NE -> "!=";
            case GT -> ">";
            case GTE -> ">=";
            case LT -> "<";
            case LTE -> "<=";
            case LIKE -> "LIKE";
            case IN -> "IN";
            case IS_NULL -> "IS NULL";
            case IS_NOT_NULL -> "IS NOT NULL";
        };
    }
}
```

### 5.2 动态数据表组件（src/components/DynamicTable/index.tsx）

```tsx
import React, { useEffect, useState } from 'react';
import { Card, Table, Space, Button, Tag, Tooltip, Typography } from 'antd';
import { explorerApi } from '@/services/explorerApi';

const { Text } = Typography;

interface ColumnMetadata {
  columnKey: string;
  columnName: string;
  displayName: string;
  dataType: string;
  renderType: string;
  width: number;
  sortable: boolean;
  filterable: boolean;
  formatConfig?: Record<string, any>;
  renderConfig?: Record<string, any>;
}

interface DynamicTableProps {
  tableKey: string;
}

export const DynamicTable: React.FC<DynamicTableProps> = ({ tableKey }) => {
  const [data, setData] = useState<Record<string, any>[]>([]);
  const [columns, setColumns] = useState<ColumnMetadata[]>([]);
  const [loading, setLoading] = useState(true);
  const [total, setTotal] = useState(0);
  const [page, setPage] = useState(0);
  const [pageSize, setPageSize] = useState(20);

  const fetchData = async () => {
    setLoading(true);
    try {
      const res = await explorerApi.executeQuery({
        tableKey,
        page,
        pageSize,
      });
      setData(res.data.data);
      setColumns(res.data.columns);
      setTotal(res.data.totalCount);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchData();
  }, [tableKey, page, pageSize]);

  const renderCell = (value: any, col: ColumnMetadata) => {
    const rt = col.renderType || getDefaultRenderType(col.dataType);

    switch (rt) {
      case 'MONEY':
        return (
          <Text>
            ¥{new Intl.NumberFormat('zh-CN').format(value)}
          </Text>
        );
      case 'TAG': {
        const colorMap = col.renderConfig?.props?.colorMap || {};
        return <Tag color={colorMap[value] || 'default'}>{value}</Tag>;
      }
      case 'LINK': {
        const href = col.renderConfig?.props?.href || '#';
        return (
          <a href={href.replace('{value}', value)} target="_blank">
            {value}
          </a>
        );
      }
      case 'DATE':
      case 'DATETIME':
        return (
          <Text type="secondary">
            {new Date(value).toLocaleString('zh-CN')}
          </Text>
        );
      }
      case 'BOOLEAN':
        return value ? '是' : '否';
      default:
        return <Text>{value ?? '-'}</Text>;
    }
  };

  const getDefaultRenderType = (dataType: string): string => {
    const map: Record<string, string> = {
      'VARCHAR': 'TEXT', 'TEXT': 'TEXT',
      'INT': 'TEXT', 'BIGINT': 'TEXT',
      'DECIMAL': 'MONEY', 'MONEY': 'MONEY',
      'DATE': 'DATE', 'DATETIME': 'DATETIME',
      'BOOLEAN': 'BOOLEAN', 'ENUM': 'TAG',
      'JSON': 'TEXT', 'HTML': 'TEXT',
    };
    return map[dataType] || 'TEXT';
  };

  const antdColumns = columns.map(col => ({
    key: col.columnName,
    dataIndex: col.columnName,
    title: col.displayName,
    width: col.width || 120,
    sortable: col.sortable,
    render: (value: any) => renderCell(value, col),
  }));

  return (
    <Card>
      <Table
        columns={antdColumns}
        dataSource={data}
        loading={loading}
        rowKey={(record) => String(record[columns.find(c => c.columnType === 'PRIMARY_KEY')?.columnName])}
        onChange={(pagination) => {
          setPage((pagination.current || 1) - 1);
          setPageSize(pagination.pageSize || 20);
        }}
        pagination={{
          current: page + 1,
          pageSize,
          total,
          showSizeChanger: true,
          showQuickJumper: true,
          showTotal: (t) => `共 ${t} 条`,
        }}
        scroll={{ x: 'max-content' }}
        size="middle"
      />
    </Card>
  );
};
```
