```java
    private final LogLlmTaskDetailRepository taskDetailRepository;
    private final LogLlmTaskStepRepository taskStepRepository;
    private final SourceViewerService sourceViewerService;

    @Override
    public List<TaskDetailDTO> getExecutionTrace(String requestId, String agentName) {
        // 1. 按 requestId（+ 可选 agentName）查 taskDetail 列表
        List<LogLlmTaskDetail> details =
            taskDetailRepository.findByRequestIdOrderByTaskIndexAsc(requestId, agentName);

        // 2. 批量查询每条 detail 的 steps
        List<Long> detailIds = details.stream()
            .map(LogLlmTaskDetail::getId)
            .toList();

        Map<Long, List<LogLlmTaskStep>> stepsMap =
            taskStepRepository.findByDetailIds(detailIds).stream()
                .collect(Collectors.groupingBy(LogLlmTaskStep::getLogTaskDetailId));

        return details.stream().map(detail -> {
            TaskDetailDTO dto = toTaskDetailDTO(detail);

            if (lineMapping != null) {
```

```java
package com.llm.insight.service.impl;

import com.llm.insight.client.LogFileClient;
import com.llm.insight.client.LogLlmAgentMainRepository;
import com.llm.insight.dto.response.*;
import com.llm.insight.service.LogViewerService;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.io.*;
import java.nio.charset.StandardCharsets;
import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;
import java.util.*;
import java.util.regex.*;
import java.util.stream.Collectors;

@Service
@RequiredArgsConstructor
public class LogViewerServiceImpl implements LogViewerService {

    private final LogFileClient logFileClient;
    private final LogLlmAgentMainRepository agentMainRepository;

    private static final int DEFAULT_PAGE_SIZE = 5000;
    private static final int LARGE_FILE_THRESHOLD = 5 * 1024 * 1024; // 5MB

    @Override
    public LogFileDTO readLogFile(String requestId, String username, int page, int pageSize) {
        String filePath = resolveLogFilePath(requestId, username);
        File file = new File(filePath);

        if (!file.exists()) {
            return LogFileDTO.notFound("日志文件不存在或已过期删除");
        }

        long fileSize = file.length();
        LogFileDTO result = new LogFileDTO();
        result.setFileName(file.getName());
        result.setFileSize(fileSize);
        result.setRequestId(requestId);

        if (fileSize > 100 * 1024 * 1024) {
            return LogFileDTO.tooLarge(fileSize);
        }

        int actualPageSize = pageSize <= 0 ? DEFAULT_PAGE_SIZE : pageSize;
        int startLine = page * actualPageSize;
        int endLine = startLine + actualPageSize;

        List<String> lines = new ArrayList<>();
        try (BufferedReader reader = new BufferedReader(
                new InputStreamReader(new FileInputStream(file), StandardCharsets.UTF_8))) {
            int lineNum = 0;
            String line;
            while ((line = reader.readLine()) != null) {
                lineNum++;
                if (lineNum > startLine && lineNum <= endLine) {
                    lines.add(line);
                }
                if (lineNum > endLine) {
                    break;
                }
            }
            result.setLines(lines);
            result.setStartLine(startLine);
            result.setEndLine(lineNum);
            result.setHasMore(lineNum > endLine);
        } catch (IOException e) {
            return LogFileDTO.error("读取日志文件失败: " + e.getMessage());
        }
        return result;
    }

    @Override
    public List<LogSearchResultDTO> searchInLog(String requestId, String username,
                                                String keyword, boolean regex) {
        String filePath = resolveLogFilePath(requestId, username);
        File file = new File(filePath);
        if (!file.exists()) {
            return List.of();
        }

        Pattern pattern = regex
            ? Pattern.compile(keyword)
            : Pattern.compile(Pattern.quote(keyword), Pattern.CASE_INSENSITIVE);

        List<LogSearchResultDTO> results = new ArrayList<>();
        try (BufferedReader reader = new BufferedReader(
                new InputStreamReader(new FileInputStream(file), StandardCharsets.UTF_8))) {
            int lineNum = 0;
            String line;
            while ((line = reader.readLine()) != null) {
                lineNum++;
                if (pattern.matcher(line).find()) {
                    results.add(LogSearchResultDTO.builder()
                        .lineNumber(lineNum)
                        .lineContent(line)
                        .build());
                    if (results.size() >= 500) { // 限制最多返回 500 条
                        break;
                    }
                }
            }
        } catch (IOException e) {
            // ignore
        }
        return results;
    }

    @Override
    public String resolveLogFilePath(String requestId, String username) {
        // 1. 从数据库获取 createTime，确定年月
        LocalDateTime createTime = agentMainRepository
            .findFirstCreateTimeByRequestId(requestId)
            .orElse(LocalDateTime.now().minusDays(1));

        String yyyyMM = createTime.format(DateTimeFormatter.ofPattern("yyyyMM"));

        // 2. 拼接路径: {dataRoot}/user/{username}/{yyyyMM}/{requestId}.log
        String dataRoot = "/app/project/data";
        return String.format("%s/user/%s/%s/%s.log", dataRoot, username, yyyyMM, requestId);
    }
}
```

```java
import com.llm.insight.client.LogLlmHttpRequestRepository;
import com.llm.insight.dto.response.*;
import com.llm.insight.service.LlmCallAnalysisService;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.util.*;

@Service
@RequiredArgsConstructor
public class LlmCallAnalysisServiceImpl implements LlmCallAnalysisService {

    private final LogLlmHttpRequestRepository httpRequestRepository;

    @Override
    public List<LlmCallDTO> listLlmCalls(String requestId) {
        return httpRequestRepository.findByRequestId(requestId).stream()
            .map(this::toDTO)
            .toList();
    }

    @Override
    public List<LlmCallDTO> getSlowCalls(String requestId, int topN) {
        return httpRequestRepository
            .findByRequestIdOrderBySpendTimeDesc(requestId, topN).stream()
            .map(this::toDTO)
            .toList();
    }

    @Override
    public TokenUsageDTO getTokenUsage(String requestId) {
        List<LogLlmHttpRequest> calls = httpRequestRepository.findByRequestId(requestId);
        int totalPrompt = calls.stream().mapToInt(c -> c.getPromptTokens() != null ? c.getPromptTokens() : 0).sum();
        int totalCompletion = calls.stream().mapToInt(c -> c.getCompletionTokens() != null ? c.getCompletionTokens() : 0).sum();

        return TokenUsageDTO.builder()
            .totalPromptTokens(totalPrompt)
            .totalCompletionTokens(totalCompletion)
            .totalTokens(totalPrompt + totalCompletion)
            .callCount(calls.size())
            .build();
    }

    @Override
    public List<LlmCallDTO> getFailedCalls(String requestId) {
        return httpRequestRepository.findByRequestIdAndSuccessExpressionFalse(requestId).stream()
            .map(this::toDTO)
            .toList();
    }

    @Override
    public LlmCallDetailDTO getCallDetail(Long callId) {
        LogLlmHttpRequest call = httpRequestRepository.findById(callId)
            .orElseThrow(() -> new IllegalArgumentException("LLM 调用记录不存在: " + callId));

        LlmCallDetailDTO dto = new LlmCallDetailDTO();
        dto.setId(call.getId());
        dto.setModelType(call.getModelType());
        dto.setSpendTime(call.getSpendTime());
        dto.setPromptTokens(call.getPromptTokens());
        dto.setCompletionTokens(call.getCompletionTokens());
        dto.setSuccessExpression(call.getSuccessExpression());
        dto.setRequestBody(call.getRequestBody());
        dto.setResponseBody(call.getResponseBody());
        dto.setCreateTime(call.getCreateTime());
        return dto;
    }
}
```

```java
package com.llm.insight.service.impl;

import com.llm.insight.dto.response.*;
import com.llm.insight.service.SourceViewerService;
import com.github.javaparser.*;
import com.github.javaparser.ast.*;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.io.*;
import java.nio.file.*;
import java.util.*;
import java.util.regex.*;

@Slf4j
@Service
public class SourceViewerServiceImpl implements SourceViewerService {

    private static final String SCRIPT_ROOT = "/app/project/data/system/agt";

    // 本地缓存：scriptPath → 解析结果
    private final Map<String, SourceScriptIndex> indexCache = new LinkedHashMap<>(100) {
        @Override
        protected boolean removeEldestEntry(Map.Entry eldest) {
            return size() > 200;
        }
    };

    @Override
    public List<ScriptFileDTO> listScripts(String agentName) {
        Path agentDir = Paths.get(SCRIPT_ROOT, agentName);
        if (!Files.exists(agentDir)) {
            return List.of();
        }
        return walkScripts(agentDir);
    }

    @Override
    public String getScriptContent(String scriptPath) {
        try {
            return Files.readString(Paths.get(scriptPath));
        } catch (IOException e) {
            log.error("读取脚本文件失败: {}", scriptPath, e);
            return "/* 脚本文件读取失败: " + scriptPath + " */";
        }
    }

    @Override
    public SourceLineMappingDTO mapTaskToLine(String agentName, String taskUniqueName) {
        // 1. 查找该 Agent 的主脚本文件
        List<ScriptFileDTO> scripts = listScripts(agentName);
        if (scripts.isEmpty()) {
            return null;
        }

        // 2. 在缓存或重新解析每个脚本文件
        for (ScriptFileDTO script : scripts) {
            SourceScriptIndex index = getOrBuildIndex(script.getFullPath());
            SourceLineRange range = index.getTaskRange(taskUniqueName);
            if (range != null) {
                return SourceLineMappingDTO.builder()
                    .agentName(agentName)
                    .taskUniqueName(taskUniqueName)
                    .filePath(script.getFullPath())
                    .startLine(range.startLine())
                    .endLine(range.endLine())
                    .build();
            }
        }
        return null;
    }

    private SourceScriptIndex getOrBuildIndex(String scriptPath) {
        return indexCache.computeIfAbsent(scriptPath, this::buildIndex);
    }

    private SourceScriptIndex buildIndex(String scriptPath) {
        SourceScriptIndex index = new SourceScriptIndex(scriptPath);
        try {
            String content = Files.readString(Paths.get(scriptPath));

            // 解析 @Plan 注解或 task() 方法
            Pattern pattern = Pattern.compile(
                "@Plan\\s*\\([^)]*uniqueName\\s*=\\s*\"([^\"]+)\"[^)]*\\)|" +
                "@Task\\s*\\([^)]*uniqueName\\s*=\\s*\"([^\"]+)\"[^)]*\\)|" +
                "task\\s*\\(\\s*\"([^\"]+)\"\\s*[,\\)]|" +
                "plan\\s*\\(\\s*\"([^\"]+)\"\\s*[,\\)]",
                Pattern.MULTILINE | Pattern.CASE_INSENSITIVE
            );

            int lineNum = 0;
            for (String line : content.split("\\r?\\n")) {
                lineNum++;
                Matcher matcher = pattern.matcher(line);
                if (matcher.find()) {
                    String uniqueName = matcher.group(1) != null ? matcher.group(1)
                                : matcher.group(2) != null ? matcher.group(2)
                                : matcher.group(3) != null ? matcher.group(3)
                                : matcher.group(4);
                    if (uniqueName != null) {
                        index.addTask(uniqueName, lineNum);
                    }
                }
            }
        } catch (IOException e) {
            log.error("解析脚本索引失败: {}", scriptPath, e);
        }
        return index;
    }

    // 内部类：脚本索引
    private record SourceLineRange(int startLine, int endLine) {}

    private static class SourceScriptIndex {
        private final String scriptPath;
        private final Map<String, SourceLineRange> taskRanges = new LinkedHashMap<>();

        SourceScriptIndex(String scriptPath) {
            this.scriptPath = scriptPath;
        }

        void addTask(String uniqueName, int startLine) {
            taskRanges.put(uniqueName, new SourceLineRange(startLine, startLine + 10));
        }

        SourceLineRange getTaskRange(String uniqueName) {
            return taskRanges.get(uniqueName);
        }
    }
}
```

```tsx
import React, { useEffect, useState, useRef } from 'react';
import MonacoEditor from '@monaco-editor/react';
import { useParams } from 'react-router-dom';

export const SourceCodeViewer: React.FC<{
  agentName: string;
  taskUniqueName: string;
  selectedTaskId?: number;
}> = ({ agentName, taskUniqueName, selectedTaskId }) => {
  const [source, setSource] = useState('');
  const [lineMapping, setLineMapping] = useState<SourceLineMappingDTO | null>(null);
  const editorRef = useRef<any>(null);

  useEffect(() => {
    // 加载源码行号映射
    requestApi.getLineMapping(agentName, taskUniqueName).then(res => {
      setLineMapping(res.data);
      if (res.data?.filePath) {
        requestApi.scriptContent(res.data.filePath).then(r => setSource(r.data));
      }
    });
  }, [agentName, taskUniqueName]);

  // 高亮选中的 task 对应的行
  useEffect(() => {
    if (editorRef.current && lineMapping) {
      editorRef.current.deltaDecorations([], [
        {
          range: {
            startLineNumber: lineMapping.startLine,
            endLineNumber: lineMapping.endLine,
          },
          options: {
            isWholeLine: true,
            className: 'highlighted-task-line',
            glyphMarginClassName: 'glyph-highlight',
          },
        },
      ]);
    }
  }, [lineMapping, selectedTaskId]);

  return (
    <div style={{ display: 'flex', height: '100vh' }}>
      {/* 左侧：源码 */}
      <div style={{ width: '50%' }}>
        <MonacoEditor
          height="100%"
          language="java"
          value={source}
          theme="vs-dark"
          onMount={(editor) => { editorRef.current = editor; }}
          options={{
            readOnly: true,
            glyphMargin: true,
            lineNumbers: 'on',
```
